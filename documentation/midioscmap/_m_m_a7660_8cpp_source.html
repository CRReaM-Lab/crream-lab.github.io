45/*Function: Read a byte from the regitster of the MMA7660*/
46uint8_t MMA7660::read(uint8_t _register) {
47 uint8_t data_read;
48 Wire.begin();
49 Wire.beginTransmission(MMA7660_ADDR);
50 Wire.write(_register);
51 Wire.endTransmission();
52 Wire.beginTransmission(MMA7660_ADDR);
53 Wire.requestFrom(MMA7660_ADDR, 1);
54 while (Wire.available()) {
55 data_read = Wire.read();

61// populate lookup table based on the MMA7660 datasheet at http://www.farnell.com/datasheets/1670762.pdf
62void MMA7660::initAccelTable() {
63 int i;
64 float val, valZ;
65
66 for (i = 0, val = 0; i < 32; i++) {
67 accLookup[i].g = val;
68 val += 0.047;
69 }
70
71 for (i = 63, val = -0.047; i > 31; i--) {
72 accLookup[i].g = val;
73 val -= 0.047;
74 }
75
76 for (i = 0, val = 0, valZ = 90; i < 22; i++) {
77 accLookup[i].xyAngle = val;
78 accLookup[i].zAngle = valZ;
79
80 val += 2.69;
81 valZ -= 2.69;
82 }
83
84 for (i = 63, val = -2.69, valZ = -87.31; i > 42; i--) {
85 accLookup[i].xyAngle = val;
86 accLookup[i].zAngle = valZ;
87
88 val -= 2.69;
89 valZ += 2.69;
90 }
91
92 for (i = 22; i < 43; i++) {
93 accLookup[i].xyAngle = 255;
94 accLookup[i].zAngle = 255;
95 }
96}

119/*Function: Get the contents of the registers in the MMA7660*/
120/* so as to calculate the acceleration. */
121bool MMA7660::getXYZ(int8_t* x, int8_t* y, int8_t* z) {
122
123START:
124
125 unsigned char val[3];
126 int count = 0;
127 val[0] = val[1] = val[2] = 64;
128 while (Wire.available() > 0) {
129 Wire.read();
130 }
131
132 Wire.requestFrom(MMA7660_ADDR, 3);
133
134 unsigned long timer_s = micros();
135
136 while (Wire.available()) {
137 if (count < 3) {
138 while (val[count] > 63) { // reload the damn thing it is bad
139 val[count] = Wire.read();
140
141 if (micros() - timer_s > MMA7660TIMEOUT) {
142 goto START;
143 }
144 }
145 }
146 count++;
147 }
148
149 *x = ((int8_t)(val[0] << 2)) / 4;
150 *y = ((int8_t)(val[1] << 2)) / 4;
151 *z = ((int8_t)(val[2] << 2)) / 4;
152
153 return 1;
154}

156bool MMA7660::getAcceleration(float* ax, float* ay, float* az) {
157 int8_t x, y, z;
158 if (!getXYZ(&x, &y, &z)) {
159 return 0;
160 }
161 *ax = x / 21.00;
162 *ay = y / 21.00;
163 *az = z / 21.00;
164
165 return 1;
166}

168bool MMA7660::getAcceleration(MMA7660_ACC_DATA* data) {
169 unsigned char val[3];
170 int count;
171 bool error;
172
173 unsigned long timer_s = micros();
174 do {
175 error = false;
176 count = 0;
177
178 while (Wire.available() > 0) {
179 Wire.read();
180 }
181
182 Wire.requestFrom(MMA7660_ADDR, 3);
183 while (Wire.available()) {
184 if (count < 3) {
185 val[count] = Wire.read();
186 if (0x40 & val[count] == 0x40) { // alert bit is set, data is garbage and we have to start over.
187 error = true;
188 break;
189 }
190 }
191 count++;
192 }
193
194 if (micros() - timer_s > MMA7660TIMEOUT) {
195 return 0;
196 }
197 } while (error);
198
199 (*data).x = accLookup[val[0]];
200 (*data).y = accLookup[val[1]];
201 (*data).z = accLookup[val[2]];
202
203 return 1;
204}

206bool MMA7660::getAllData(MMA7660_DATA* data) {
207 int count = 0;
208 uint8_t val[11] = {0};
209
210 while (Wire.available() > 0) {
211 Wire.read();
212 }
213
214 Wire.requestFrom(MMA7660_ADDR, 11);
215 while (Wire.available()) {
216 if (count < 11) {
217 val[count] = Wire.read();
218 }
219 count++;
220 }
221
222 data->X = val[0];
223 data->Y = val[1];
224 data->Z = val[2];
225 data->TILT = val[3];
226 data->SRST = val[4];
227 data->SPCNT = val[5];
228 data->INTSU = val[6];
229 data->MODE = val[7];
230 data->SR = val[8];
231 data->PDET = val[9];
232 data->PD = val[10];
233
234 return 1;
